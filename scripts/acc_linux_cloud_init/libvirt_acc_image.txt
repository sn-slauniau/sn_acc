# Set up a MID server and validate it on the instance
# Enable ACC listener; if using port 8443 for testing, run the following to forward traffic from 443 to 8443:

sudo firewall-cmd --add-forward-port=port=443:proto=tcp:toport=8443
success

# Create a MID Web Server API Key and keep the value in your notes

export AGENT_RPM=agent-client-collector-6.0.4-x86_64.rpm
export MY_MID_HOSTNAME=mid.internal
read -s MY_API_KEY
echo $MY_API_KEY

cp -a settings.conf.template settings.conf
sed -i "s/AGENT_RPM/${AGENT_RPM}/" settings.conf
sed -i "s/MY_MID_HOSTNAME/${MY_MID_HOSTNAME}/" settings.conf
sed -i "s/MY_API_KEY/${MY_API_KEY}/" settings.conf
cat settings.conf

export VDI_USER=alma
read -s VDI_PWD
export VDI_PWD

# download the official image and make a copy of it
curl -O --silent https://repo.almalinux.org/almalinux/10/cloud/x86_64/images/AlmaLinux-10-GenericCloud-latest.x86_64.qcow2
cp -a AlmaLinux-10-GenericCloud-latest.x86_64.qcow2 alma10-base.qcow2

# verify the checksum
curl -O --silent https://repo.almalinux.org/almalinux/10/cloud/x86_64/images/CHECKSUM
cat CHECKSUM
sha256sum alma10-base.qcow2 
8729862a9e2d93c478c2334b33d01d7ee21934a795a9b68550cc1c63ffd0efae  alma10-base.qcow2

# install ACC on the base image
sudo -E virt-customize -a alma10-base.qcow2 --network --upload settings.conf:/tmp/settings.conf --run install_sn_acc.sh 2>/dev/null

        <name>agent-client-collector</name>
